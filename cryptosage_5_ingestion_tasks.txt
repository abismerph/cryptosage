# ── Load source config ────────────────────────────────────────────────────────
def load_sources() -> list[dict]:
    config_path = Path(settings.NEWS_SOURCES_CONFIG_PATH)
    if not config_path.exists():
        log.warning("sources_config_not_found", path=str(config_path))
        return _default_sources()
    with open(config_path) as f:
        return json.load(f)

# ── Article Fetcher ───────────────────────────────────────────────────────────
async def fetch_rss_feed(source: dict, session: aiohttp.ClientSession) -> list[dict[str, Any]]:
    """Parse an RSS feed and return a list of raw article dicts."""
    articles = []
    try:
        async with session.get(source["url"], timeout=aiohttp.ClientTimeout(total=15)) as resp:
            text = await resp.text()
        feed = feedparser.parse(text)
        for entry in feed.entries[:30]:  # cap per source
            published = None
            if hasattr(entry, "published_parsed") and entry.published_parsed:
                published = datetime(*entry.published_parsed[:6], tzinfo=timezone.utc)

            content = ""
            if hasattr(entry, "content"):
                content = entry.content[0].get("value", "")
            elif hasattr(entry, "summary"):
                content = entry.summary

            # Strip HTML tags from content
            soup = BeautifulSoup(content, "html.parser")
            content_clean = soup.get_text(separator=" ", strip=True)

            articles.append({
                "title":       getattr(entry, "title", "").strip(),
                "url":         getattr(entry, "link", ""),
                "content":     content_clean[:8000],
                "source_name": source["name"],
                "author":      getattr(entry, "author", None),
                "image_url":   _extract_image(entry),
                "published_at": published,
            })
    except Exception as exc:
        log.warning("rss_fetch_error", source=source["name"], error=str(exc))
    return articles


def _extract_image(entry) -> str | None:
    """Try to extract a thumbnail/image URL from a feed entry."""
    if hasattr(entry, "media_thumbnail") and entry.media_thumbnail:
        return entry.media_thumbnail[0].get("url")
    if hasattr(entry, "enclosures") and entry.enclosures:
        for enc in entry.enclosures:
            if enc.get("type", "").startswith("image/"):
                return enc.get("href")
    return None


# ── Content Hash ──────────────────────────────────────────────────────────────
def compute_hash(title: str, url: str) -> str:
    return hashlib.sha256(f"{title.lower().strip()}|{url.strip()}".encode()).hexdigest()[:32]


# ── Ingestion Pipeline ────────────────────────────────────────────────────────
async def ingest_all_sources() -> dict[str, int]:
    """
    Main ingestion entry point. Run by the Celery beat scheduler.
    Returns stats: {fetched, new, skipped, errors}.
    """
    sources = load_sources()
    stats = {"fetched": 0, "new": 0, "skipped": 0, "errors": 0}

    connector = aiohttp.TCPConnector(limit=20)
    async with aiohttp.ClientSession(
        connector=connector,
        headers={"User-Agent": "CryptoSage-Bot/1.0 (https://cryptosage.ai)"},
    ) as http_session:

        # Fetch all feeds concurrently
        tasks = [fetch_rss_feed(src, http_session) for src in sources]
        results = await asyncio.gather(*tasks, return_exceptions=True)

    all_articles: list[dict] = []
    for result in results:
        if isinstance(result, list):
            all_articles.extend(result)
        else:
            stats["errors"] += 1

    stats["fetched"] = len(all_articles)
    log.info("ingestion_fetched", count=len(all_articles), sources=len(sources))

    # Process & store
    async with AsyncSessionLocal() as db:
        for raw in all_articles[:settings.MAX_ARTICLES_PER_RUN]:
            try:
                saved = await _process_and_save(raw, db)
                if saved:
                    stats["new"] += 1
                else:
                    stats["skipped"] += 1
            except Exception as exc:
                log.warning("article_save_error", error=str(exc), title=raw.get("title", "")[:60])
                stats["errors"] += 1
        await db.commit()

    log.info("ingestion_complete", **stats)
    return stats


async def _process_and_save(raw: dict, db: AsyncSession) -> bool:
    """
    Normalise, deduplicate, tag, score, and persist one article.
    Returns True if newly saved, False if duplicate.
    """
    title = raw.get("title", "").strip()
    url   = raw.get("url", "").strip()
    if not title or not url:
        return False

    content_hash = compute_hash(title, url)

    # Deduplication check
    existing = await db.execute(
        select(Article.id).where(Article.content_hash == content_hash)
    )
    if existing.scalar_one_or_none() is not None:
        return False   # duplicate

    # Coin tagging
    content_text = f"{title} {raw.get('content', '')}"
    coins_mentioned = tag_coins(content_text)

    # Importance score
    importance = score_importance(title, raw.get("content", ""), coins_mentioned)

    article = Article(
        title=title,
        url=url,
        content=raw.get("content"),
        source_name=raw.get("source_name", "Unknown"),
        author=raw.get("author"),
        image_url=raw.get("image_url"),
        coins_mentioned=coins_mentioned,
        content_hash=content_hash,
        importance_score=importance,
        published_at=raw.get("published_at"),
    )
    db.add(article)
    return True

    "DAI":  [r"\bdai\b"],
}

# Pre-compile patterns for performance
_COMPILED: dict[str, list[re.Pattern]] = {
    symbol: [re.compile(p, re.IGNORECASE) for p in patterns]
    for symbol, patterns in COIN_REGISTRY.items()
}


def tag_coins(text: str, max_coins: int = 10) -> list[str]:
    """
    Extract coin ticker symbols mentioned in `text`.
    Returns a deduplicated list of uppercase ticker strings, ordered by first appearance.

    Args:
        text:      Article title + body text (concatenated)
        max_coins: Maximum number of coins to return (prevents spam articles)
    """
    if not text:
        return []

    found: list[str] = []
    seen: set[str] = set()

    for symbol, patterns in _COMPILED.items():
        for pattern in patterns:
            if pattern.search(text):
                if symbol not in seen:
                    found.append(symbol)
                    seen.add(symbol)
                break  # found for this symbol, move on

        if len(found) >= max_coins:
            break

    return found


def is_crypto_related(text: str) -> bool:
    """
    Quick heuristic to check if an article is crypto-related.
    Used to filter out false positives from general finance feeds.
    """
    CRYPTO_KEYWORDS = [
        "crypto", "blockchain", "defi", "nft", "token", "wallet",
        "ethereum", "bitcoin", "solana", "web3", "smart contract",
        "exchange", "coinbase", "binance", "altcoin", "stablecoin",
        "mining", "staking", "yield", "dao", "layer 2", "layer2",
        "on-chain", "dex", "cex", "whitepaper", "tokenomics",
    ]
    text_lower = text.lower()
    return any(kw in text_lower for kw in CRYPTO_KEYWORDS)

"""
CryptoSage AI — Importance Scorer
Heuristic-based article importance scoring (0.0–1.0).
Fast, deterministic, no LLM calls. Used at ingestion time.
"""
import re

# ── Signal Weights ────────────────────────────────────────────────────────────
# Each matched keyword adds its weight to the raw score (capped at 1.0)

HIGH_IMPACT_TERMS = {
    # Regulatory / legal
    "sec": 0.25, "cftc": 0.25, "regulation": 0.20, "ban": 0.20, "law": 0.15,
    "lawsuit": 0.20, "court": 0.15, "congress": 0.20, "senate": 0.18,
    "doj": 0.25, "arrest": 0.25, "fraud": 0.22, "sanction": 0.22,
    # Market events
    "etf": 0.20, "ipo": 0.18, "listing": 0.15, "delisting": 0.22,
    "hack": 0.30, "exploit": 0.30, "breach": 0.28, "stolen": 0.28,
    "bankruptcy": 0.30, "insolvency": 0.28, "collapse": 0.28, "rug pull": 0.30,
    "partnership": 0.10, "acquisition": 0.18, "merger": 0.18,
    "billion": 0.10, "million": 0.06,
    # Major institutions
    "blackrock": 0.20, "fidelity": 0.18, "jp morgan": 0.18, "goldman": 0.18,
    "coinbase": 0.12, "binance": 0.12, "federal reserve": 0.18, "fed": 0.10,
    # Protocol events
    "upgrade": 0.12, "fork": 0.15, "mainnet": 0.12, "launch": 0.08,
    "halving": 0.20, "merge": 0.15,
    # Price action
    "all-time high": 0.18, "ath": 0.18, "crash": 0.22, "rally": 0.10,
    "liquidation": 0.18, "flash crash": 0.25,
}

TIER1_COINS = {"BTC", "ETH", "SOL", "BNB", "XRP", "ADA", "AVAX"}
TIER2_COINS = {"DOT", "MATIC", "LINK", "UNI", "ARB", "OP", "NEAR", "ATOM"}

TRUSTED_SOURCES = {
    "coindesk", "the block", "decrypt", "cointelegraph",
    "bitcoin magazine", "cryptoslate", "reuters", "bloomberg",
    "financial times", "wall street journal",
}


def score_importance(
    title: str,
    content: str,
    coins_mentioned: list[str],
    source_name: str = "",
) -> float:
    """
    Compute an importance score for an article (0.0–1.0).

    Factors:
      - High-impact keyword matches in title (weighted 2x) and content
      - Tier of coins mentioned
      - Source credibility
      - Article length (proxy for depth)

    Returns:
        float in [0.0, 1.0]
    """
    score = 0.0
    title_lower   = title.lower()
    content_lower = (content or "").lower()

    # ── Keyword scoring ───────────────────────────────────────────────────────
    for term, weight in HIGH_IMPACT_TERMS.items():
        in_title   = term in title_lower
        in_content = term in content_lower
        if in_title:
            score += weight * 2.0   # title match is 2× more significant
        elif in_content:
            score += weight * 0.5

    # ── Coin tier bonus ───────────────────────────────────────────────────────
    if any(c in TIER1_COINS for c in coins_mentioned):
        score += 0.10
    elif any(c in TIER2_COINS for c in coins_mentioned):
        score += 0.05

    # Multiple coins → broader market relevance
    if len(coins_mentioned) >= 4:
        score += 0.05

    # ── Source credibility bonus ──────────────────────────────────────────────
    source_lower = source_name.lower()
    if any(ts in source_lower for ts in TRUSTED_SOURCES):
        score += 0.05

    # ── Length signal ─────────────────────────────────────────────────────────
    word_count = len((content or "").split())
    if word_count > 800:
        score += 0.05
    elif word_count > 300:
        score += 0.02

    # ── Clamp to [0.0, 1.0] ───────────────────────────────────────────────────
    return round(min(score, 1.0), 4)

"""
CryptoSage AI — Celery Task: News Ingestion
Triggered every 2 hours by Celery Beat.
"""
import asyncio
import structlog
from app.tasks.celery_app import celery_app

"""
CryptoSage AI — Celery Task: Embedding Pipeline
Picks up newly ingested articles (is_embedded=False) and upserts them into Pinecone.
Runs every 2h + 15min, after the ingestion task completes.
"""
import asyncio
import structlog
from sqlalchemy import select, update

from app.tasks.celery_app import celery_app
from app.core.database import AsyncSessionLocal
from app.models.article import Article
from app.services.ai.rag import upsert_batch

log = structlog.get_logger(__name__)

BATCH_SIZE = 50


@celery_app.task(
    name="app.tasks.embed_articles.embed_uningested_articles",
    bind=True,
    max_retries=3,
    default_retry_delay=120,
    soft_time_limit=900,
    time_limit=960,
)
def embed_uningested_articles(self):
    """
    Celery task: fetches articles with is_embedded=False from PostgreSQL,
    generates embeddings in batches, upserts into Pinecone,
    and marks articles as embedded.
    """
    log.info("embedding_task_started")

    async def _run():
        total_embedded = 0
        async with AsyncSessionLocal() as db:
            # Fetch unembedded articles in batches
            result = await db.execute(
                select(Article)
                .where(Article.is_embedded == False)
                .where(Article.content.isnot(None))
                .order_by(Article.ingested_at.asc())
                .limit(BATCH_SIZE * 10)  # process up to 500 per run
            )
            articles = list(result.scalars().all())

            if not articles:
                log.info("embedding_task_no_new_articles")
                return 0

            log.info("embedding_task_articles_to_embed", count=len(articles))

            # Process in batches of BATCH_SIZE
            for i in range(0, len(articles), BATCH_SIZE):
                batch = articles[i : i + BATCH_SIZE]
                records = [
                    {
                        "id": str(a.id),
                        "title": a.title,
                        "content": (a.content or "")[:8000],
                        "metadata": {
                            "source_name": a.source_name,
                            "coins_mentioned": a.coins_mentioned or [],
                            "sentiment": a.sentiment or "neutral",
                            "published_at": a.published_at.isoformat() if a.published_at else "",
                            "url": a.url,
                        },
                    }
                    for a in batch
                ]

                try:
                    count = await upsert_batch(records)
                    # Mark as embedded
                    article_ids = [a.id for a in batch]
                    await db.execute(
                        update(Article)
                        .where(Article.id.in_(article_ids))
                        .values(
                            is_embedded=True,
                            pinecone_id=None,  # batch upsert sets IDs internally
                        )
                    )
                    await db.commit()
                    total_embedded += count
                    log.info("embedding_batch_complete", batch=i // BATCH_SIZE + 1, count=count)

                except Exception as exc:
                    log.error("embedding_batch_failed", batch=i // BATCH_SIZE + 1, error=str(exc))
                    # Continue with next batch — don't fail the whole task

        return total_embedded

    try:
        total = asyncio.get_event_loop().run_until_complete(_run())
        log.info("embedding_task_complete", total_embedded=total)
        return {"embedded": total}
    except Exception as exc:
        log.error("embedding_task_failed", error=str(exc))
        raise self.retry(exc=exc)

"""
CryptoSage AI — Celery Task: Morning Brief™ Email Digest
Runs at 8am UTC daily. Generates daily summary and emails subscribed users.
"""
import asyncio
import structlog
from sqlalchemy import select

from app.tasks.celery_app import celery_app
from app.core.database import AsyncSessionLocal
from app.models.user import User
from app.services.ai.summarizer import get_daily_summary
from app.services.digest.email_sender import send_morning_brief

log = structlog.get_logger(__name__)

BATCH_SIZE = 100   # Send emails in batches to avoid rate limits


@celery_app.task(
    name="app.tasks.send_digests.send_daily_digest",
    bind=True,
    max_retries=2,
    default_retry_delay=600,
    soft_time_limit=1800,
    time_limit=1860,
)
def send_daily_digest(self):
    """
    Celery task: generates the daily digest and emails all subscribed users.
    Respects each user's digest_frequency and email_digest_enabled flag.
    """
    log.info("digest_task_started")

    async def _run():
        async with AsyncSessionLocal() as db:
            # Generate digest once — reused for all users
            summary = await get_daily_summary(db)

            # Fetch subscribed users (daily digest only)
            result = await db.execute(
                select(User)
                .where(User.is_active == True)
                .where(User.email_digest_enabled == True)
                .where(User.digest_frequency == "daily")
            )
            users = list(result.scalars().all())

        log.info("digest_task_users", count=len(users))
        sent = 0
        failed = 0

        # Send in batches
        for i in range(0, len(users), BATCH_SIZE):
            batch = users[i : i + BATCH_SIZE]
            for user in batch:
                try:
                    await send_morning_brief(user=user, summary=summary)
                    sent += 1
                except Exception as exc:
                    log.warning("digest_send_failed", user=str(user.id), error=str(exc))
                    failed += 1

        return {"sent": sent, "failed": failed, "total_users": len(users)}

    try:
        stats = asyncio.get_event_loop().run_until_complete(_run())
        log.info("digest_task_complete", **stats)
        return stats
    except Exception as exc:
        log.error("digest_task_failed", error=str(exc))
        raise self.retry(exc=exc)


# END: app/tasks/send_digests.py
