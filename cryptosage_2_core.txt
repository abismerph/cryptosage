@app.middleware("http")
async def add_request_id(request: Request, call_next):
    import uuid
    request_id = request.headers.get("X-Request-ID", str(uuid.uuid4())[:8])
    with structlog.contextvars.bound_contextvars(request_id=request_id):
        response = await call_next(request)
    response.headers["X-Request-ID"] = request_id
    return response

# ── Dependency ────────────────────────────────────────────────────────────────
async def get_db() -> AsyncGenerator[AsyncSession, None]:
    """
    FastAPI dependency that yields an async DB session.
    Usage:
        @router.get("/")
        async def my_endpoint(db: AsyncSession = Depends(get_db)):
            ...
    """
    async with AsyncSessionLocal() as session:
        try:
            yield session
            await session.commit()
        except Exception:
            await session.rollback()
            raise
        finally:
            await session.close()

"""
CryptoSage AI — Redis Client & Cache Utilities
Provides async Redis connection pool and cache helpers.
"""
import json
from typing import Any, Optional
import redis.asyncio as aioredis
from app.core.config import settings

# ── Connection Pool ───────────────────────────────────────────────────────────
redis_pool = aioredis.ConnectionPool.from_url(
    settings.REDIS_URL,
    max_connections=20,
    decode_responses=True,
)


def get_redis() -> aioredis.Redis:
    """Return a Redis client bound to the shared connection pool."""
    return aioredis.Redis(connection_pool=redis_pool)


# ── Cache Helpers ─────────────────────────────────────────────────────────────
async def cache_get(key: str) -> Optional[Any]:
    """Fetch a JSON-serialised value from Redis. Returns None on miss."""
    client = get_redis()
    raw = await client.get(key)
    if raw is None:
        return None
    return json.loads(raw)


async def cache_set(key: str, value: Any, ttl: int) -> None:
    """Store a JSON-serialised value in Redis with TTL in seconds."""
    client = get_redis()
    await client.set(key, json.dumps(value), ex=ttl)


async def cache_delete(key: str) -> None:
    """Invalidate a single cache key."""
    client = get_redis()
    await client.delete(key)


async def cache_incr(key: str, ttl: int = 86400) -> int:
    """
    Atomic increment — used for daily usage counters.
    Sets TTL on first write so the key auto-expires at midnight.
    Returns the new count.
    """
    client = get_redis()
    count = await client.incr(key)
    if count == 1:
        await client.expire(key, ttl)
    return count


# ── Key Builders ──────────────────────────────────────────────────────────────
def key_daily_summary() -> str:
    from datetime import date
    return f"summary:daily:{date.today().isoformat()}"


def key_coin_feed(coin: str, page: int = 1) -> str:

def key_sentiment(coin: str) -> str:
    from datetime import datetime
    hour = datetime.utcnow().strftime("%Y%m%d%H")
    return f"sentiment:{coin.lower()}:{hour}"


def key_explain(query_hash: str) -> str:

def key_usage_count(user_id: str, feature: str) -> str:

def key_rate_limit(user_id: str) -> str:

"""
CryptoSage AI — Authentication Middleware
Verifies Clerk JWT tokens and resolves the current user from the database.
"""
from typing import Optional
import httpx
from fastapi import Depends, HTTPException, Security, status
from fastapi.security import HTTPAuthorizationCredentials, HTTPBearer
from jose import JWTError, jwt
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.core.config import settings
from app.core.database import get_db
from app.models.user import User, PlanTier

# ── Clerk JWKS ────────────────────────────────────────────────────────────────
_JWKS_URL = "https://api.clerk.com/v1/jwks"
_jwks_cache: dict = {}

bearer_scheme = HTTPBearer(auto_error=False)


async def _get_clerk_public_keys() -> dict:
    """Fetch and cache Clerk's JSON Web Key Set."""
    if _jwks_cache:
        return _jwks_cache
    async with httpx.AsyncClient() as client:
        resp = await client.get(
            _JWKS_URL,
            headers={"Authorization": f"Bearer {settings.CLERK_SECRET_KEY}"},
            timeout=10,
        )
        resp.raise_for_status()
        _jwks_cache.update(resp.json())
    return _jwks_cache


async def _decode_clerk_token(token: str) -> dict:
    """Decode and validate a Clerk-issued JWT. Returns the claims payload."""
    try:
        jwks = await _get_clerk_public_keys()
        header = jwt.get_unverified_header(token)
        # Find the matching key by kid
        key = next(
            (k for k in jwks.get("keys", []) if k.get("kid") == header.get("kid")),
            None,
        )
        if key is None:
            raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Invalid token key")

        payload = jwt.decode(
            token,
            key,
            algorithms=["RS256"],
            options={"verify_aud": False},
        )
        return payload
    except JWTError as exc:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail=f"Token validation failed: {exc}",
        )


# ── Core Dependency ───────────────────────────────────────────────────────────
async def get_current_user(
    credentials: Optional[HTTPAuthorizationCredentials] = Security(bearer_scheme),
    db: AsyncSession = Depends(get_db),
) -> User:
    """
    FastAPI dependency: validates Bearer token, returns the User ORM object.
    Raises 401 if token is missing/invalid, 404 if user not in DB.
    """
    if credentials is None:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Authentication required",
            headers={"WWW-Authenticate": "Bearer"},
        )

    payload = await _decode_clerk_token(credentials.credentials)
    clerk_user_id: str = payload.get("sub", "")

    if not clerk_user_id:
        raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Invalid token subject")

    result = await db.execute(select(User).where(User.clerk_user_id == clerk_user_id))
    user = result.scalar_one_or_none()

    if user is None:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="User not found — complete onboarding first")

    if not user.is_active:
        raise HTTPException(status_code=status.HTTP_403_FORBIDDEN, detail="Account suspended")

    return user


async def get_optional_user(
    credentials: Optional[HTTPAuthorizationCredentials] = Security(bearer_scheme),
    db: AsyncSession = Depends(get_db),
) -> Optional[User]:
    """Same as get_current_user but returns None instead of raising for unauthenticated requests."""
    if credentials is None:
        return None
    try:
        return await get_current_user(credentials, db)
    except HTTPException:
        return None


# ── Plan Guards ───────────────────────────────────────────────────────────────
def require_plan(*plans: PlanTier):
    """
    Factory that returns a dependency enforcing a minimum plan tier.

    Usage:
        @router.get("/pro-feature")
        async def pro_only(user: User = Depends(require_plan(PlanTier.PRO))):
            ...
    """

                detail=f"This feature requires one of: {[p.value for p in plans]}. "
                       f"Upgrade your plan at cryptosage.ai/pricing",
            )
        return user
    return _guard

"""
CryptoSage AI — Rate Limiting & Feature Gate Middleware
Enforces per-plan API rate limits and daily usage quotas via Redis.
"""
import time
from fastapi import Depends, HTTPException, status
from sqlalchemy.ext.asyncio import AsyncSession

from app.core.config import settings
from app.core.database import get_db
from app.core.redis import cache_incr, key_rate_limit, key_usage_count
from app.core.auth import get_current_user
from app.models.user import User, PlanTier


# ── Rate Limit Map ────────────────────────────────────────────────────────────
RATE_LIMITS: dict[PlanTier, int] = {
    PlanTier.FREE:       settings.RATE_LIMIT_FREE,
    PlanTier.PRO:        settings.RATE_LIMIT_PRO,
    PlanTier.ENTERPRISE: settings.RATE_LIMIT_ENTERPRISE,
}

# ── Daily Usage Limits ────────────────────────────────────────────────────────
DAILY_LIMITS: dict[str, dict[PlanTier, int]] = {
    "explain": {
        PlanTier.FREE:       settings.FREE_EXPLAIN_LIMIT,
        PlanTier.PRO:        999_999,
        PlanTier.ENTERPRISE: 999_999,
    },
    "whitepaper": {
        PlanTier.FREE:       0,
        PlanTier.PRO:        settings.PRO_WHITEPAPER_LIMIT,
        PlanTier.ENTERPRISE: 999_999,
    },
    "api": {
        PlanTier.FREE:       0,
        PlanTier.PRO:        settings.PRO_API_CALLS_LIMIT,
        PlanTier.ENTERPRISE: 999_999,
    },
}


async def check_rate_limit(user: User) -> None:
    """
    Enforce per-minute rate limit based on plan tier.
    Uses Redis INCR with 60-second TTL.
    """
    limit = RATE_LIMITS.get(user.plan, settings.RATE_LIMIT_FREE)
    key = key_rate_limit(str(user.id))

    # Redis client handles TTL atomically via cache_incr
    count = await cache_incr(key, ttl=60)
    if count > limit:
        raise HTTPException(
            status_code=status.HTTP_429_TOO_MANY_REQUESTS,
            detail=f"Rate limit exceeded ({limit} req/min for {user.plan.value} plan). "
                   f"Retry after 60 seconds.",
            headers={"Retry-After": "60"},
        )


async def check_daily_limit(user: User, feature: str) -> int:
    """
    Enforce daily usage quota per feature per plan.
    Returns the updated count.
    """
    limits = DAILY_LIMITS.get(feature, {})
    limit = limits.get(user.plan, 0)

    if limit == 0:
        raise HTTPException(
            status_code=status.HTTP_403_FORBIDDEN,
            detail=f"Feature '{feature}' is not available on the {user.plan.value} plan. "
                   f"Upgrade at cryptosage.ai/pricing",
        )

    key = key_usage_count(str(user.id), feature)
    count = await cache_incr(key, ttl=86400)  # resets every 24 hours

    if count > limit:
        raise HTTPException(
            status_code=status.HTTP_429_TOO_MANY_REQUESTS,
            detail=f"Daily limit reached: {limit} {feature} calls/day on {user.plan.value} plan. "
                   f"Resets at midnight UTC. Upgrade at cryptosage.ai/pricing",
            headers={"X-Daily-Limit": str(limit), "X-Daily-Used": str(count)},
        )

    return count


# ── Composite Guard ───────────────────────────────────────────────────────────
def feature_guard(feature: str):
    """
    FastAPI dependency factory combining rate limit + daily limit checks.

    Usage:
        @router.post("/explain")
        async def explain(
            user: User = Depends(feature_guard("explain")),
            ...
        ):
    """
    async def _guard(
        user: User = Depends(get_current_user),
        db: AsyncSession = Depends(get_db),
    ) -> User:
        await check_rate_limit(user)
        await check_daily_limit(user, feature)
        return user

    return _guard
