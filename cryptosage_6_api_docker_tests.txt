"""
CryptoSage AI — /api/v1/feed  (ClarityFeed™ coin feed)
"""
from fastapi import APIRouter, Depends, Path, Query
from sqlalchemy import select, func
from sqlalchemy.ext.asyncio import AsyncSession

from app.core.database import get_db
from app.core.auth import get_current_user
from app.core.redis import cache_get, cache_set, key_coin_feed
from app.core.config import settings
from app.models.article import Article
from app.models.user import User
from app.schemas.feed import FeedResponse, ArticleSchema

router = APIRouter(prefix="/feed", tags=["ClarityFeed — News Feed"])

PAGE_SIZE = 20


@router.get(
    "",
    response_model=FeedResponse,
    summary="Get the latest crypto news feed (all coins)",
)
async def global_feed(
    page: int = Query(1, ge=1, description="Page number"),
    user: User = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),
) -> FeedResponse:
    """
    Returns the global news feed across all coins, ordered by importance
    and publication date. Cached per page for 15 minutes.
    """
    return await _paginated_feed(db, coin=None, page=page)


@router.get(
    "/{coin}",
    response_model=FeedResponse,
    summary="Get coin-specific news feed",
)
async def coin_feed(
    coin: str = Path(..., min_length=1, max_length=16, description="Coin ticker, e.g. BTC"),
    page: int = Query(1, ge=1),
    user: User = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),
) -> FeedResponse:
    """
    Returns news feed filtered to a specific coin.
    Ordered by importance score then publication date descending.
    Cached per coin/page for 15 minutes.
    """
    return await _paginated_feed(db, coin=coin.upper(), page=page)


async def _paginated_feed(db: AsyncSession, coin: str | None, page: int) -> FeedResponse:
    """Shared pagination logic for global and coin-specific feeds."""
    cache_key = key_coin_feed(coin or "global", page)
    cached = await cache_get(cache_key)
    if cached:
        return FeedResponse(**cached)

    offset = (page - 1) * PAGE_SIZE
    base_query = select(Article)

    if coin:
        base_query = base_query.where(Article.coins_mentioned.contains([coin]))

    # Count total
    count_q = select(func.count()).select_from(base_query.subquery())
    total = (await db.execute(count_q)).scalar_one()

    # Fetch page
    result = await db.execute(
        base_query
        .order_by(Article.importance_score.desc().nulls_last(), Article.published_at.desc())
        .offset(offset)
        .limit(PAGE_SIZE)
    )
    articles = list(result.scalars().all())

    response_data = dict(
        coin=coin,
        articles=[
            ArticleSchema(
                id=str(a.id),
                title=a.title,
                url=a.url,
                source_name=a.source_name,
                summary=a.summary,
                coins_mentioned=a.coins_mentioned or [],
                sentiment=a.sentiment,
                sentiment_score=a.sentiment_score,
                importance_score=a.importance_score,
                published_at=a.published_at,
                image_url=a.image_url,
            ).model_dump(mode="json")
            for a in articles
        ],
        page=page,
        page_size=PAGE_SIZE,
        total=total,
        has_next=offset + PAGE_SIZE < total,
    )

    await cache_set(cache_key, response_data, ttl=900)  # 15-min cache
    return FeedResponse(**response_data)

from fastapi import APIRouter, Depends, HTTPException, Path, status
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.core.database import get_db
from app.core.auth import get_current_user
from app.core.config import settings
from app.models.user import User, PlanTier
from app.models.watchlist import WatchlistItem
from app.schemas.watchlist import WatchlistCreate, WatchlistResponse

router = APIRouter(prefix="/watchlist", tags=["Watchlist"])


async def _get_watchlist_count(user_id, db: AsyncSession) -> int:
    result = await db.execute(
        select(WatchlistItem).where(WatchlistItem.user_id == user_id)
    )
    return len(result.scalars().all())


@router.get("", response_model=list[WatchlistResponse], summary="Get user's watchlist")
async def get_watchlist(
    user: User = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),
) -> list[WatchlistResponse]:
    result = await db.execute(
        select(WatchlistItem)
        .where(WatchlistItem.user_id == user.id)
        .order_by(WatchlistItem.sort_order, WatchlistItem.created_at)
    )
    return list(result.scalars().all())


@router.post("", response_model=WatchlistResponse, status_code=status.HTTP_201_CREATED,
             summary="Add coin to watchlist")
async def add_to_watchlist(
    payload: WatchlistCreate,
    user: User = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),
) -> WatchlistResponse:
    # Plan-tier coin limit
    limit = settings.FREE_WATCHLIST_LIMIT if user.plan == PlanTier.FREE else 999
    current_count = await _get_watchlist_count(user.id, db)
    if current_count >= limit:
        raise HTTPException(
            status_code=status.HTTP_403_FORBIDDEN,
            detail=f"Watchlist limit reached ({limit} coins on {user.plan.value} plan). "
                   f"Upgrade at cryptosage.ai/pricing",
        )

    # Upsert (ignore duplicate coin_symbol per user via unique constraint)
    item = WatchlistItem(
        user_id=user.id,
        coin_symbol=payload.coin_symbol.upper(),
        coin_name=payload.coin_name,
        coingecko_id=payload.coingecko_id,
        alert_on_major_news=payload.alert_on_major_news,
        alert_on_sentiment_shift=payload.alert_on_sentiment_shift,
        include_in_digest=payload.include_in_digest,
        sort_order=current_count,
    )
    db.add(item)
    await db.flush()
    await db.refresh(item)
    return item


@router.delete("/{coin_symbol}", status_code=status.HTTP_204_NO_CONTENT,
               summary="Remove coin from watchlist")
async def remove_from_watchlist(
    coin_symbol: str = Path(..., min_length=1, max_length=16),
    user: User = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),
) -> None:
    result = await db.execute(
        select(WatchlistItem).where(
            WatchlistItem.user_id == user.id,
            WatchlistItem.coin_symbol == coin_symbol.upper(),
        )
    )
    item = result.scalar_one_or_none()
    if item is None:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Coin not in watchlist")
    await db.delete(item)

"""
CryptoSage AI — /api/v1/webhooks/stripe
Handles Stripe subscription lifecycle events to sync plan tier in DB.
"""
import stripe
from fastapi import APIRouter, Depends, HTTPException, Request, status
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession
import structlog

from app.core.config import settings
from app.core.database import get_db
from app.models.user import User, PlanTier

router = APIRouter(prefix="/webhooks", tags=["Webhooks"])
log = structlog.get_logger(__name__)

stripe.api_key = settings.STRIPE_SECRET_KEY

# Map Stripe price IDs → plan tiers
PRICE_TO_PLAN: dict[str, PlanTier] = {
    settings.STRIPE_PRICE_ID_PRO_MONTHLY:   PlanTier.PRO,
    settings.STRIPE_PRICE_ID_PRO_ANNUAL:    PlanTier.PRO,
    settings.STRIPE_PRICE_ID_ENTERPRISE:    PlanTier.ENTERPRISE,
}


@router.post("/stripe", summary="Stripe webhook receiver", include_in_schema=False)
async def stripe_webhook(request: Request, db: AsyncSession = Depends(get_db)) -> dict:
    """
    Receives and processes Stripe webhook events.
    Verifies the signature using STRIPE_WEBHOOK_SECRET.
    """
    payload = await request.body()
    sig_header = request.headers.get("stripe-signature", "")

    try:
        event = stripe.Webhook.construct_event(
            payload, sig_header, settings.STRIPE_WEBHOOK_SECRET
        )
    except stripe.error.SignatureVerificationError as exc:
        log.warning("stripe_signature_invalid", error=str(exc))
        raise HTTPException(status_code=status.HTTP_400_BAD_REQUEST, detail="Invalid signature")

    event_type = event["type"]
    data = event["data"]["object"]
    log.info("stripe_webhook_received", type=event_type)

    # ── Subscription created / updated ────────────────────────────────────────
    if event_type in ("customer.subscription.created", "customer.subscription.updated"):
        await _handle_subscription_change(data, db, active=True)

    # ── Subscription cancelled / deleted ──────────────────────────────────────
    elif event_type in ("customer.subscription.deleted",):
        await _handle_subscription_change(data, db, active=False)

    # ── Payment failed ────────────────────────────────────────────────────────
    elif event_type == "invoice.payment_failed":
        customer_id = data.get("customer")
        log.warning("stripe_payment_failed", customer=customer_id)
        # Optionally: send dunning email here

    return {"status": "ok"}


async def _handle_subscription_change(
    subscription: dict,
    db: AsyncSession,
    active: bool,
) -> None:
    """Sync Stripe subscription state to the User table."""
    customer_id: str = subscription.get("customer", "")
    subscription_id: str = subscription.get("id", "")
    price_id: str = ""

    items = subscription.get("items", {}).get("data", [])
    if items:
        price_id = items[0].get("price", {}).get("id", "")

    result = await db.execute(
        select(User).where(User.stripe_customer_id == customer_id)
    )
    user = result.scalar_one_or_none()
    if user is None:
        log.warning("stripe_customer_not_found", customer=customer_id)
        return

    if active:
        plan = PRICE_TO_PLAN.get(price_id, PlanTier.FREE)
        user.plan = plan
        user.stripe_subscription_id = subscription_id
        log.info("user_plan_upgraded", user=str(user.id), plan=plan.value)
    else:
        user.plan = PlanTier.FREE
        user.stripe_subscription_id = None
        log.info("user_plan_downgraded", user=str(user.id))

    await db.flush()

"""
CryptoSage AI — /api/v1/webhooks/clerk
Handles Clerk user lifecycle events to sync users into our DB.
"""
import hmac
import hashlib
import json
from fastapi import APIRouter, Depends, HTTPException, Request, status
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession
import structlog

from app.core.config import settings
from app.core.database import get_db
from app.models.user import User

router = APIRouter(prefix="/webhooks", tags=["Webhooks"])
log = structlog.get_logger(__name__)


def _verify_clerk_signature(payload: bytes, svix_id: str, svix_ts: str, svix_sig: str) -> bool:
    """Verify Clerk webhook signature using HMAC-SHA256."""
    to_sign = f"{svix_id}.{svix_ts}.{payload.decode()}".encode()
    secret = settings.CLERK_WEBHOOK_SECRET.replace("whsec_", "")
    import base64
    key = base64.b64decode(secret)
    expected = base64.b64encode(
        hmac.new(key, to_sign, hashlib.sha256).digest()
    ).decode()
    return any(sig.split(",")[1] == expected for sig in svix_sig.split(" "))


@router.post("/clerk", summary="Clerk webhook receiver", include_in_schema=False)
async def clerk_webhook(request: Request, db: AsyncSession = Depends(get_db)) -> dict:
    """
    Receives Clerk user lifecycle events (created, updated, deleted).
    Syncs user data into the CryptoSage users table.
    """
    payload = await request.body()
    svix_id  = request.headers.get("svix-id", "")
    svix_ts  = request.headers.get("svix-timestamp", "")
    svix_sig = request.headers.get("svix-signature", "")

    if not _verify_clerk_signature(payload, svix_id, svix_ts, svix_sig):
        log.warning("clerk_signature_invalid")
        raise HTTPException(status_code=status.HTTP_400_BAD_REQUEST, detail="Invalid signature")

    event = json.loads(payload)
    event_type: str = event.get("type", "")
    data: dict = event.get("data", {})

    log.info("clerk_webhook_received", type=event_type)

    if event_type == "user.created":
        await _upsert_user(data, db)

    elif event_type == "user.updated":
        await _upsert_user(data, db)

    elif event_type == "user.deleted":
        clerk_id = data.get("id", "")
        result = await db.execute(select(User).where(User.clerk_user_id == clerk_id))
        user = result.scalar_one_or_none()
        if user:
            user.is_active = False
            await db.flush()
            log.info("user_deactivated", clerk_id=clerk_id)

    return {"status": "ok"}


async def _upsert_user(data: dict, db: AsyncSession) -> User:
    """Create or update a user from Clerk event data."""
    clerk_id = data.get("id", "")
    emails = data.get("email_addresses", [])
    primary_email_id = data.get("primary_email_address_id")
    email = ""
    for e in emails:
        if e.get("id") == primary_email_id:
            email = e.get("email_address", "")
            break
    if not email and emails:
        email = emails[0].get("email_address", "")

    first = data.get("first_name") or ""
    last  = data.get("last_name") or ""
    full_name = f"{first} {last}".strip() or None
    avatar_url = data.get("image_url")

    result = await db.execute(select(User).where(User.clerk_user_id == clerk_id))
    user = result.scalar_one_or_none()

    if user is None:
        user = User(
            clerk_user_id=clerk_id,
            email=email,
            full_name=full_name,
            avatar_url=avatar_url,
        )
        db.add(user)
        log.info("user_created", clerk_id=clerk_id, email=email)
    else:
        user.email = email or user.email
        user.full_name = full_name or user.full_name
        user.avatar_url = avatar_url or user.avatar_url
        log.info("user_updated", clerk_id=clerk_id)

    await db.flush()
    return user

Supports both sync (offline) and async (online) migration modes.

import asyncio

from alembic import context

# Import all models so Alembic can detect schema changes

config = context.config

if config.config_file_name is not None:
    fileConfig(config.config_file_name)

target_metadata = Base.metadata


def run_migrations_offline() -> None:
    url = config.get_main_option("sqlalchemy.url")
    context.configure(
        url=url,
        target_metadata=target_metadata,
        literal_binds=True,
        dialect_opts={"paramstyle": "named"},
        compare_type=True,
    )
    with context.begin_transaction():
        context.run_migrations()


def do_run_migrations(connection: Connection) -> None:
    context.configure(
        connection=connection,
        target_metadata=target_metadata,
        compare_type=True,
    )
    with context.begin_transaction():
        context.run_migrations()


async def run_async_migrations() -> None:
    connectable = async_engine_from_config(
        config.get_section(config.config_ini_section, {}),
        prefix="sqlalchemy.",
        poolclass=pool.NullPool,
    )
    async with connectable.connect() as connection:
        await connection.run_sync(do_run_migrations)
    await connectable.dispose()


def run_migrations_online() -> None:
    asyncio.run(run_async_migrations())


if context.is_offline_mode():
    run_migrations_offline()
else:
    run_migrations_online()

"""
CryptoSage AI — Tests: Coin Tagger & Importance Scorer

import pytest
from app.services.ingestion.coin_tagger import tag_coins, is_crypto_related
from app.services.ingestion.importance_scorer import score_importance


class TestCoinTagger:

    def test_tags_bitcoin_by_name(self):
        coins = tag_coins("Bitcoin hits new all-time high as ETF flows surge")
        assert "BTC" in coins

    def test_tags_ethereum_ticker(self):
        coins = tag_coins("ETH gas fees drop dramatically after upgrade")
        assert "ETH" in coins

    def test_tags_multiple_coins(self):
        coins = tag_coins("BTC and ETH lead the market rally alongside SOL")
        assert "BTC" in coins
        assert "ETH" in coins
        assert "SOL" in coins

    def test_case_insensitive(self):
        coins = tag_coins("solana network experiences brief outage")
        assert "SOL" in coins

    def test_empty_text_returns_empty(self):
        assert tag_coins("") == []

    def test_no_crypto_text_returns_empty(self):
        coins = tag_coins("The weather today is sunny and warm in Denver.")
        assert coins == []

    def test_max_coins_respected(self):
        text = "BTC ETH SOL BNB XRP ADA AVAX DOT MATIC LINK UNI AAVE ARB OP"
        coins = tag_coins(text, max_coins=5)
        assert len(coins) <= 5

    def test_deduplicates_mentions(self):
        coins = tag_coins("Bitcoin is rising. Bitcoin breaks $100k. BTC volume surges.")
        assert coins.count("BTC") == 1


class TestCryptoRelated:

    def test_crypto_article_detected(self):
        assert is_crypto_related("DeFi protocol launches new yield farming product") is True

    def test_non_crypto_article_rejected(self):
        assert is_crypto_related("The stock market closed higher today on positive jobs data") is False

    def test_blockchain_keyword_triggers(self):
        assert is_crypto_related("Major bank explores blockchain for settlements") is True


class TestImportanceScorer:

    def test_hack_scores_high(self):
        score = score_importance("Major DeFi hack drains $200 million", "", ["ETH"])
        assert score >= 0.50

    def test_general_article_scores_lower(self):
        score = score_importance("Weekend crypto roundup: markets quiet", "", [])
        assert score < 0.30

    def test_sec_regulation_scores_high(self):
        score = score_importance("SEC sues major exchange over unregistered securities", "", ["BTC"])
        assert score >= 0.40

    def test_tier1_coin_bonus_applied(self):
        score_btc = score_importance("Price update", "", ["BTC"])
        score_none = score_importance("Price update", "", [])
        assert score_btc > score_none

    def test_score_capped_at_one(self):
        # Extremely high-signal article should not exceed 1.0
        score = score_importance(
            "SEC sues Binance for fraud hack bankruptcy exploit billion",
            "blackrock fidelity arrest lawsuit collapse",
            ["BTC", "ETH", "SOL"],
        )
        assert score <= 1.0

    def test_score_is_float(self):
        score = score_importance("Test title", "Test content", [])
        assert isinstance(score, float)
